import random
import time

streak = 0
rando1 = random.randint(1,9)
while True:

    print("Will the next number be higher or lower than " + str(rando1) +"?")
    guess = input()
    if guess.capitalize()[0] == 'Q':
        break
    while((guess.capitalize()[0] != 'H') and (guess.capitalize()[0] != 'L') ):

        print("Invalid input, try again with H or L")
        guess = input()
    rando2 = random.randint(1,9)
    print("The next number is")
    time.sleep(1)
    print("." , end="")
    time.sleep(1)
    print(".",end="")
    time.sleep(1)
    print(".")
    time.sleep(1)
    print(str(rando2) + "!")

    if (((rando2 > rando1) and (guess.capitalize()[0] == 'H')) or ((rando2 < rando1) and (guess.capitalize()[0] == 'L'))):
        streak +=1
        print("You Win!!   Your win streak is: " + str(streak))
        rando1 = rando2

    elif rando2 == rando1:
        rando1 = random.randint(1, 9)
        print("The numbers where the same! You don't win, but you don't lose your streak Your win streak is: " + str(streak))
        continue

    else:
        print("You Lose!!    Your win streak was: " + str(streak))
        streak = 0
        rando1 = random.randint(1, 9)

print("Goodbye!")
